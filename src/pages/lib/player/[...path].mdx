import { useSSG } from "nextra/ssg";
import { getVideoData } from "@/utils/getList";
import Artplayer from "@/components/Artplayer";
import artplayerPluginDanmuku from "artplayer-plugin-danmuku";
import { decode } from "js-base64";
import { useState } from "react";

export function getStaticPaths() {
  //this property is an Array of URLs ("paths") that should be statically generated at build-time. The returned paths must match the dynamic route shape.
  return {
    paths: [],
    fallback: true,
  };
}

export const getStaticProps = async ({ params }) => {
  const path = params.path;
  if (
    !path ||
    !path[0] ||
    !path[1] ||
    !path[2] ||
    !path[3] ||
    path[0] == ("0" || 0) ||
    path[1] == ("0" || 0) ||
    path[2] == ("0" || 0) ||
    path[3] == ("0" || 0)
  )
    return {
      props: {
        ssg: {
          dlinks: { video: "", danmaku: "", sub_chs: "", sub_cht: "" },
          co: {
            video: "unknown",
            danmaku: false,
            sub_chs: false,
            sub_cht: false,
          },
        },
      },
      revalidate: 10,
    };
  else {
    const a = await getVideoData(path[0], path[1], path[2], path[3]);
    return {
      props: {
        ssg: {
          dlinks: a,
          co: JSON.parse(
            decode(
              path[3] ||
                "eyJ2aWRlbyI6InVua25vd24iLCJkYW5tYWt1IjpmYWxzZSwic3ViX2NocyI6ZmFsc2UsInN1Yl9jaHQiOmZhbHNlfQ=="
            )
          ),
        },
      },
      revalidate: 60,
    };
  }
};

export function AP() {
  const { dlinks, co } = useSSG() || {
    dlinks: { video: "", danmaku: "", sub_chs: "", sub_cht: "" },
    co: {
      video: "unknown",
      danmaku: false,
      sub_chs: false,
      sub_cht: false,
    },
  };
  const [showMore, setShowMore] = useState(false);
  function handleMoreClick() {
    setShowMore(!showMore);
  }
  if (!dlinks || !dlinks.video) return <div>加载中.../获取失败?</div>;
  else {
    const sm = co.video === "mp4" || showMore;
    return (
      <div>
        {!sm &&
          "当前视频可能无法在浏览器正常播放，请下载 弹弹Play 本地播放。"}
        <br />
        {!sm && (
          <button onClick={handleMoreClick}>
            点击强制播放(可能黑屏、无声、无字幕)
          </button>
        )}
        {sm && (
          <Artplayer
            option={{
              url: dlinks.video,
              autoSize: true,
              playbackRate: true,
              aspectRatio: true,
              setting: true,
              hotkey: true,
              mutex: true,
              fullscreen: true,
              fullscreenWeb: true,
              fastForward: true,
              whitelist: ["*"],
              subtitleOffset: true,
              miniProgressBar: true,
              subtitle:
                dlinks.sub_chs || dlinks.sub_cht
                  ? {
                      url: dlinks.sub_chs || dlinks.sub_cht,
                      type: "srt",
                      encoding: "utf-8",
                      style: {
                        color: "#FFFFFF",
                        background: "rgba(0, 0, 0, 0.4)",
                        border: "1px solid rgba(0,0,0,0)",
                        "border-radius": "5px",
                        width: "300px",
                        "margin-inline": "31.5%",
                      },
                    }
                  : {},
              plugins: dlinks.danmaku
                ? [
                    artplayerPluginDanmuku({
                      danmuku: dlinks.danmaku,
                    }),
                  ]
                : [],
            }}
            style={{
              width: "500px",
              height: "300px",
              margin: "30px auto 0",
            }}
            getInstance={(art) => console.info(art)}
          />
        )}
      </div>
    );
  }
}
